	scope = character
	is_shown = {
		OR = {
			light_magic_lifestyle_perks > 0
			shadow_magic_lifestyle_perks > 0
			disorder_magic_lifestyle_perks > 0
			order_magic_lifestyle_perks > 0
			life_magic_lifestyle_perks > 0
			death_magic_lifestyle_perks > 0

			order_magic_lifestyle_perk_points > 0
			life_magic_lifestyle_perk_points > 0

			elemental_magic_lifestyle_perk_points > 0
		}
	}

	}
	effect = {
		wc_regenerate_mana_effect = yes
	}
}
